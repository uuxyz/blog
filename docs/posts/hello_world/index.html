Here, I showcase several ways to write a “Hello world” program in some of my favorite programming languages.

scheme

(display "Hello world!")

python

print("Hello world!")

rust

fn main() {
   print!("Hello world!");
}

c

#include<stdio.h>
int main(){
  printf("\nHello world!");
  return 0;
}

c++

import module std;
int main() {
  std::print("Hello world!\n");
}

java

public class Main {
    public static void main(String[] args) {
        System.out.println("Hello, World!");
    }
}
